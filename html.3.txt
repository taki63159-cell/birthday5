function openGift(){

"💖 Dear Anwesha 💖 <br> May your birthday be filled with happiness, laughter and beautiful memories. You deserve all the joy in the world! 🎂✨";

/* confetti */

function confetti(){

for(let i=0;i<60;i++){

let c=document.createElement("div");

c.style.position="absolute";

c.style.width="8px";

c.style.height="8px";

c.style.background="hsl("+Math.random()*360+"deg,100%,50%)";

c.style.left=Math.random()*100+"%";

c.style.top="-10px";

document.body.appendChild(c);

let fall=setInterval(()=>{

c.style.top=(c.offsetTop+5)+"px";

if(c.offsetTop>window.innerHeight){

clearInterval(fall);

c.remove();

}

},20);

}

}
